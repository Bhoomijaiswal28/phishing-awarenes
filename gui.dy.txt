import tkinter as tk
from tkinter import messagebox

from phishing_detector import analyze_message


def analyze():
    msg = text.get("1.0", tk.END).strip()

    if not msg:
        messagebox.showwarning("Warning", "Please enter a message.")
        return

    result = analyze_message(msg)

    output.config(state="normal")
    output.delete("1.0", tk.END)
    output.insert(tk.END, result)
    output.config(state="disabled")


def clear():
    text.delete("1.0", tk.END)
    output.config(state="normal")
    output.delete("1.0", tk.END)
    output.config(state="disabled")


root = tk.Tk()
root.title("Phishing Awareness Detector")
root.geometry("700x550")

tk.Label(root, text="Phishing Awareness Detector",
         font=("Arial", 18, "bold")).pack(pady=10)

tk.Label(root, text="Enter Email / SMS").pack()

text = tk.Text(root, width=80, height=12)
text.pack(pady=5)

tk.Button(root, text="Analyze",
          command=analyze,
          width=15).pack(pady=5)

tk.Button(root, text="Clear",
          command=clear,
          width=15).pack()

tk.Label(root, text="Result").pack(pady=5)

output = tk.Text(root, width=80, height=10, state="disabled")
output.pack()

root.mainloop()
